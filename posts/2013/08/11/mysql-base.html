mysql命令行

help或\h获得帮助  help select

use

show

use db_name;
show tables;

列出表的所有列信息

show columns from table_name;
or
desc table_name;

show create database db_name;
show create table table_name;

其他

show status  服务器状态信息
show grants  显示授权用户
show errors/show warnings 显示服务器错误或警告信息

select

检索单个列

select prod_name from products;

多个列

select prod_id, prod_name, prod_price from products;

检索所有列

select * from products;
#除非确认要用到所有列

检索去重

select distinct prod_id from products;

限制结果数

select prod_id from products limit 5;

select prod_id from products limit 5, 5;
                             limit 5 offset 5;

order by

按某个字段排序

select prod_name from products order by prod_name;

按多列排序

select prod_id, prod_name from products order by prod_id, prod_price;

指定排序方向（升序降序）

select prod_id, prod_name from products order by prod_id desc;
#默认升序asc

where

过滤

select prod_id from products where prod_price = 2.5;

过滤不匹配

select prod_id from products where prod_price <> 2.5;

范围检查

select prod_id from products where prod_price between 2.5 and 5;

空值检查

select prod_id from products where prod_prin is null;

多条件，组合and

select prod_id from products where prod_price = 2.5 and vend_id = 1000;

多条件, 组合or

select prod_id from products where prod_price = 2.5 or vend_id = 1000;

优先级 and 大于 or, 先处理的and,所以应该适当使用括号

select prod_id from products where (prod_price < 2.5 or vend_id = 1000) and prod_price > 1;

指定查询范围, in操作符

select prod_id from products where vend_id in (1001, 1002);

取反，not操作符

select prod_id from products where vend_id not in (1001, 1002);

通配

select prod_id from products where prod_name like 'jet%'

单个字符

select prod_id from products where prod_name like 'j_t%'

regexp

正则搜索

select prod_id from products where prod_name regexp '1000';
select prod_id from products where prod_name regexp '.000';

进行or匹配

select prod_id from products where prod_name regexp '1000|2000';

几个之一

匹配范围

select prod_id from products where prod_name regexp '[1-5]000';

匹配特殊字符，\ 进行转义

select prod_id from products where prod_name regexp '\\.';

like和 regexp

like整列匹配
regexp 列值内匹配

concat

拼接字符

select concat(vend_name, '-', vend_id) from vendors;

select concat(trim(vend_name), '-', vend_id) from vendors;

select concat(trim(vend_name), '-', vend_id) as vend_title from vendors;

聚集函数

avg 平均

select avg(prod_price) from products;

count 计数

select count(*) from products; #无论Null还是非空，均纳入计数
select count(prod_id) from products; #计数有值记录，忽略NULL值

max 最大

select max(prod_price) from products;

min 最小

select min(prod_price) from products;

sum 求和

#sum函数忽略值为NULL的行

group

group

select vend_id, count(*) as num_prods from products group by vend_id;

注意：

1.group by 可以包含任意数目的列
2.group by 中每个列都必须是检索列或有效的表达式（但不能使聚集函数）
3.除聚集函数外，select语句中的每个列都必须在group by子句中出现
4.如果分组列有Null值，Null将作为一个分组返回
5.group by 子句必须出现在where子句之后, order by 之前

过滤分组

select cust_id from orders group by cust_id having count(*) >= 2;

where和having区别

where在分组前过滤，having在分组后过滤
